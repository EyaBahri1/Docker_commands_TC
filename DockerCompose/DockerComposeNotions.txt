---

Docker Compose is a file that allows us to run multiple containers together.

1. Install Docker Compose.

2. Run `docker-compose --help` or `docker-compose --version` to test the commands.

- Unlike changing a Dockerfile, if you modify the `docker-compose.yml` file, you don't need to rebuild the images.

---

Main commands:

```bash
# Start all services defined in the file
docker-compose up

# Start in detached mode (in the background)
docker-compose up -d

# Stop all services
docker-compose down

# Recreate and restart services (rebuild if needed)
docker-compose up --build

# See the status of containers
docker-compose ps

# View logs
docker-compose logs

# View logs in real-time
docker-compose logs -f

# Stop a specific service
docker-compose stop <service_name>

# Restart a specific service
docker-compose restart <service_name>

# Remove containers, networks, and volumes (be careful)
docker-compose down -v
```

---

Example:

version: '3'

services:
  backend:
    container_name: mern-backend
    build: ./backend
    ports:
      - "5000:5000"
    environment:
      - NODE_ENV=development
    depends_on:
      - mongo
    command: npm start
    volumes:
      - ./backend:/app

  frontend:
    container_name: mern-frontend
    build: ./frontend
    ports:
      - "3000:3000"
    environment:
      - REACT_APP_API_URL=http://localhost:5000
    depends_on:
      - backend
    command: npm start
    volumes:
      - ./frontend:/app

  mongo:
    container_name: mern-mongo
    image: mongo:latest
    ports:
      - "27017:27017"
    volumes:
      - mongo-data:/data/db

volumes:
  mongo-data:
```
