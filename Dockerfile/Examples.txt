**Example 2 with Node.js**

# FROM: base image with Node.js
FROM node:14

# WORKDIR: working directory
WORKDIR /app

# COPY: copy package.json and package-lock.json
COPY package*.json ./

# RUN: install dependencies
RUN npm install

# COPY: copy the rest of the code
COPY . .

# ENV: environment variable (optional)
ENV PORT 80

# EXPOSE: port used by the Node.js app
EXPOSE ${PORT}

# CMD: start the application
CMD ["npm", "start"]
